df = bldg_sfha

pre <- df |> mutate(new_info_released = 2017) |>
  filter(sale_year < 2020)

# Compare slopes over time by SFHA, by High FF, and by their interaction
m_sfha   <- feols(log_price ~ sale_year * in_eff_sfha, data = pre, cluster = ~pin10)
m_highff <- feols(log_price ~ sale_year * high_ff_score, data = pre, cluster = ~pin10)

# technically high_ff_Score is not known yet, but it could could convey true risk and if properties that were high risk were already on a different trend than less risky properties.

modelsummary(

  output     = "gt",
